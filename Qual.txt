function Players(func, includenewplayers)
    for i,v in pairs(Services.Players:GetPlayers()) do
        func(v)
    end
	if includenewplayers then
		Services.Players.PlayerAdded:connect(func)
	end
end

function IsEnemy(v)
	return v.Team ~= Player.Team
end

function IsAlly(v)
	return v.Team == Player.Team
end

function Enemies(func)
	Players(function(v)
		if IsEnemy(v) then
        	func(v)
		end
	end)
